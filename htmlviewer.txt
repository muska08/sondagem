        // Carregar dados salvos
        function loadPollData() {
            const saved = JSON.parse(localStorage.getItem('footballPollData') || '{"votes": {}, "totalVotes": 0}');
            pollData = saved;
            
            // Inicializar votos para todas as quest√µes se n√£o existirem
            questions.forEach(question => {
                if (!pollData.votes[question.id]) {
                    pollData.votes[question.id] = {};
                    question.options.forEach(option => {
                        pollData.votes[question.id][option] = 0;
                    });
                }
            });
        }

        // Salvar dados
        function savePollData() {
            localStorage.setItem('footballPollData', JSON.stringify(pollData));
        }

        // Submeter formul√°rio
        document.getElementById('pollForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const questions = ['q1', 'q2', 'q3', 'q4', 'q5', 'q6', 'q7', 'q8', 'q9', 'q10', 'q11', 'q12'];
            let allAnswered = true;
            
            questions.forEach(question => {
                const radios = document.querySelectorAll(`input[name="${question}"]`);
                const checked = Array.from(radios).some(radio => radio.checked);
                if (!checked) {
                    allAnswered = false;
                }
            });
            
            if (!allAnswered) {
                alert('Por favor, responda a todas as perguntas antes de submeter.');
                return;
            }
            
            // Recolher respostas
            const formData = new FormData(this);
            const answers = {};
            
            for (let [key, value] of formData.entries()) {
                answers[key] = value;
                // Incrementar voto
                pollData.votes[key][value]++;
            }
            
            pollData.totalVotes++;
            savePollData();
            
            // Mostrar resultados
            showResults();
        });

        // Mostrar resultados
        function showResults() {
            document.getElementById('pollSection').style.display = 'none';
            document.getElementById('resultsSection').style.display = 'block';
            
            const resultsContent = document.getElementById('resultsContent');
            resultsContent.innerHTML = '';
            
            questions.forEach((question, index) => {
                const questionVotes = pollData.votes[question.id];
                const totalQuestionVotes = Object.values(questionVotes).reduce((sum, votes) => sum + votes, 0);
                
                const resultDiv = document.createElement('div');
                resultDiv.className = 'result-item';
                
                const option1 = question.options[0];
                const option2 = question.options[1];
                const votes1 = questionVotes[option1] || 0;
                const votes2 = questionVotes[option2] || 0;
                
                const percentage1 = totalQuestionVotes > 0 ? Math.round((votes1 / totalQuestionVotes) * 100) : 0;
                const percentage2 = totalQuestionVotes > 0 ? Math.round((votes2 / totalQuestionVotes) * 100) : 0;
                
                const winner1 = votes1 > votes2;
                const winner2 = votes2 > votes1;
                
                resultDiv.innerHTML = `
                    <div class="result-title">${question.title}</div>
                    <div class="result-comparison">
                        <div class="result-player ${winner1 ? 'winner' : ''}">
                            <div class="result-player-name">${option1}</div>
                            <div class="result-percentage count-animation">${percentage1}%</div>
                            <div class="result-bar">
                                <div class="result-bar-fill" style="width: ${percentage1}%"></div>
                            </div>
                        </div>
                        <div class="result-player ${winner2 ? 'winner' : ''}">
                            <div class="result-player-name">${option2}</div>
                            <div class="result-percentage count-animation">${percentage2}%</div>
                            <div class="result-bar">
                                <div class="result-bar-fill" style="width: ${percentage2}%"></div>
                            </div>
                        </div>
                    </div>
                `;
                
                resultsContent.appendChild(resultDiv);
            });
            
            // Mostrar total de votos
            document.getElementById('totalVotes').innerHTML = `
                <h3>üìà Total de Participantes</h3>
                <p style="font-size: 2em; font-weight: bold; margin-top: 10px;">${pollData.totalVotes}</p>
                <p>Obrigado pela sua participa√ß√£o!</p>
            `;
            
            // Animar as barras
            setTimeout(() => {
                document.querySelectorAll('.result-bar-fill').forEach(bar => {
                    bar.style.transition = 'width 1.5s ease';
                });
            }, 100);
        }
